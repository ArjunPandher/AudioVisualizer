Basic Audio Visualiser Canvas

A basic canvas that displays an animation that syncs up to the music being played.

Song Name: I'm God - Clams Casino
  • JS
  • HTML
  • CSS
const audioVisGen = new AudioVisGenerator("player")
const barCanvas = document.getElementById("canvas-bar")
barCanvas.width = "1000"
barCanvas.height = "300"
const barCanvasVis = audioVisGen.addCanvas(barCanvas, 0, "#000000", "#ffb4a2")
const player1 = document.getElementById("player1")
let playerHasBeenPlayed = false
player1.onplay = () => {
  if (!playerHasBeenPlayed) {
   audioVisGen.setUp()
   playerHasBeenPlayed = true
 }
 audioVisGen.animateAllCanvases()
}
<audio controls id="player1">
 <source src="Im%20God.mp3" type="audio/mp3">
</audio>
<div id="audioVisDiv">
 <canvas id="canvas-bar" width="1000" height="300"></canvas>
</div>
#canvas-bar {
  background-color: #ffb4a2;
}
